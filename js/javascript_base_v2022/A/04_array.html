Array

        1.배열생성
        let 배열명 = []; //빈배열 생성
        배열명[0] = 값;
        배열명[1] = 값;

        2.배열생성
        let 배열명 = [값, 값, 값]; 

        배열명[인덱스번호]  //출력

length

배열의 개수

join

        배열->문자열 변환

        배열명.join();
        배열명.join('/'); 배열의 값들을 /로 구분하여 하나의 문자열로 합친다

reverse

        배열을 역순으로(원본 수정)
        배열.reverse();

sort

        오름, 내림, 알바벳, 가나다...(원본수정)
        배열명.sort()//알파벳순, 오름차순

push

        배열의 마지막에 값 추가
        배열.push('새값');

pop

        배열의 마지막에 값 제거
        배열.pop();

unshift

        배열의 첫번째 추가
        배열.unshift('새값');

shift

        첫번째 값 제거
        배열.shift();

slice

        인덱스번호로 원소값을 추출(복사)
        배열.slice(시작, 끝); 시작번호 부터 끝번호 앞자리까지

splice

        인덱스번호로 원소값을 추출(오려내기)
        배열.splice(시작, 끝); 시작번호 부터 끝번호 앞자리까지

concat

        배열과 배열을 합체
        배열.concat(배열);  //합칠대상 - 배열

filter

        조건에 맞는 값만 새 배열로 생성

        배열.filter(function(매개변수){
            return 매개변수...
        })

        매개변수는 배열의 각 값을 대변할 변수
